# Alert

The Alert component is used to display important messages or feedback to users. It supports different types of alerts for various scenarios like success, error, warning, and info messages.

## Import

```tsx
import { Alert } from '@lazy-panda-ui/lazy-panda-ui';
```

## Usage

### Basic Alert

```tsx
import { Alert } from '@lazy-panda-ui/lazy-panda-ui';

function Example() {
  return (
    <Alert message="This is a basic info alert" />
  );
}
```

### Different Types

```tsx
import { Alert } from '@lazy-panda-ui/lazy-panda-ui';

function Example() {
  return (
    <>
      <Alert 
        type="success"
        message="Operation completed successfully"
      />
      
      <Alert 
        type="error"
        message="An error occurred"
      />
      
      <Alert 
        type="warning"
        message="Please proceed with caution"
      />
      
      <Alert 
        type="info"
        message="Here's some useful information"
      />
    </>
  );
}
```

### Dismissible Alert

```tsx
import { Alert } from '@lazy-panda-ui/lazy-panda-ui';

function Example() {
  const handleDismiss = () => {
    console.log('Alert dismissed');
  };

  return (
    <Alert
      message="This alert can be dismissed"
      type="info"
      dismissible
      onDismiss={handleDismiss}
    />
  );
}
```

### With Custom Icon

```tsx
import { Alert } from '@lazy-panda-ui/lazy-panda-ui';
import { Icon } from 'your-icon-library';

function Example() {
  return (
    <Alert
      message="Alert with custom icon"
      icon={<Icon name="bell" />}
      type="info"
    />
  );
}
```

## Props

### Main Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `message` | string | Required | The message to display |
| `type` | 'success' \| 'info' \| 'warning' \| 'error' | 'info' | The type of alert |
| `icon` | React.ReactNode | - | Custom icon to display |
| `dismissible` | boolean | false | Whether the alert can be dismissed |
| `onDismiss` | () => void | - | Called when alert is dismissed |

### Style Props

| Prop | Type | Description |
|------|------|-------------|
| `containerStyle` | ViewStyle | Additional styles for the container |
| `contentStyle` | ViewStyle | Additional styles for the content wrapper |
| `messageStyle` | TextStyle | Additional styles for the message text |
| `iconStyle` | ViewStyle | Additional styles for the icon wrapper |

### Accessibility Props

| Prop | Type | Description |
|------|------|-------------|
| `accessibilityRole` | AccessibilityRole | Role for screen readers |
| `accessibilityLabel` | string | Accessibility label |
| `testID` | string | Test ID for testing |

## Theme Customization

You can customize the Alert component through the theme:

```tsx
const theme = {
  components: {
    alert: {
      container: {
        // base container styles
      },
      content: {
        // content wrapper styles
      },
      iconContainer: {
        // icon wrapper styles
      },
      text: {
        // message text styles
      },
      dismissButton: {
        // dismiss button styles
      },
      dismissIcon: {
        // dismiss icon styles
      },
      variants: {
        success: {
          container: {
            backgroundColor: 'rgba(52, 199, 89, 0.1)',
            borderColor: 'rgba(52, 199, 89, 0.3)',
          },
          text: {
            color: '#2A7D47',
          },
        },
        // Similar customizations for info, warning, and error variants
      },
    },
  },
};
```
