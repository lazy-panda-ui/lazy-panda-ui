# Accordion

The Accordion component helps create collapsible content sections, ideal for FAQs, settings menus, or any content that needs to be organized in expandable sections.

## Import

```tsx
import { Accordion } from '@lazy-panda-ui/lazy-panda-ui';
```

## Usage

### Basic Accordion

```tsx
import { Accordion } from '@lazy-panda-ui/lazy-panda-ui';

function Example() {
  return (
    <Accordion
      title="What is Lazy Panda UI?"
      subtitle="Click to learn more"
    >
      <Text>
        Lazy Panda UI is a modern React Native component library designed
        for building beautiful cross-platform applications.
      </Text>
    </Accordion>
  );
}
```

### Controlled Accordion

```tsx
import { Accordion } from '@lazy-panda-ui/lazy-panda-ui';

function Example() {
  const [expanded, setExpanded] = useState(false);

  return (
    <Accordion
      title="Controlled Accordion"
      expanded={expanded}
      onChange={setExpanded}
    >
      <Text>This accordion's state is controlled externally.</Text>
    </Accordion>
  );
}
```

### Different Variants

```tsx
import { Accordion } from '@lazy-panda-ui/lazy-panda-ui';

function Example() {
  return (
    <>
      <Accordion
        title="Default Variant"
        variant="default"
      >
        <Text>Content for default variant</Text>
      </Accordion>

      <Accordion
        title="Outlined Variant"
        variant="outlined"
      >
        <Text>Content for outlined variant</Text>
      </Accordion>

      <Accordion
        title="Contained Variant"
        variant="contained"
      >
        <Text>Content for contained variant</Text>
      </Accordion>
    </>
  );
}
```

### With Custom Icons

```tsx
import { Accordion } from '@lazy-panda-ui/lazy-panda-ui';

function Example() {
  return (
    <Accordion
      title="Custom Icons"
      expandIcon={<Icon name="chevron-down" />}
      collapseIcon={<Icon name="chevron-up" />}
      iconPosition="start"
    >
      <Text>Accordion with custom expand/collapse icons</Text>
    </Accordion>
  );
}
```

## Props

### Main Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `title` | string | Required | Title of the accordion |
| `subtitle` | string | - | Optional subtitle text |
| `children` | React.ReactNode | Required | Content to be displayed when expanded |
| `expanded` | boolean | - | Controls the expanded state (controlled mode) |
| `onChange` | (expanded: boolean) => void | - | Called when accordion is toggled |
| `variant` | 'default' \| 'outlined' \| 'contained' | 'default' | Visual variant of the accordion |
| `iconPosition` | 'start' \| 'end' | 'end' | Position of the expand/collapse icon |
| `disabled` | boolean | false | Whether the accordion is disabled |

### Style Props

| Prop | Type | Description |
|------|------|-------------|
| `containerStyle` | ViewStyle | Additional styles for the container |
| `headerStyle` | ViewStyle | Additional styles for the header |
| `titleStyle` | TextStyle | Additional styles for the title |
| `subtitleStyle` | TextStyle | Additional styles for the subtitle |
| `contentStyle` | ViewStyle | Additional styles for the content |

### Accessibility Props

| Prop | Type | Description |
|------|------|-------------|
| `testID` | string | Test ID for testing |
| `accessibilityLabel` | string | Accessibility label for screen readers |
| `accessibilityHint` | string | Additional description for screen readers |

## Theme Customization

You can customize the Accordion component through the theme:

```tsx
const theme = {
  components: {
    accordion: {
      container: {
        // container styles
      },
      header: {
        // header styles
      },
      title: {
        // title text styles
      },
      subtitle: {
        // subtitle text styles
      },
      content: {
        // content wrapper styles
      },
      variants: {
        outlined: {
          // outlined variant styles
        },
        contained: {
          // contained variant styles
        },
      },
    },
  },
};
```
